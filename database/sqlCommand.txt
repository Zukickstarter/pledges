CREATE TABLE IF NOT EXISTS `pledgeOptions` (
  `id` INTEGER auto_increment ,
  `price` INTEGER,
  `pledgeTitle` VARCHAR(255),
  `description` VARCHAR(255),
  `estDelivery` VARCHAR(255),
  `backers` INTEGER DEFAULT 0,
  `listingId` INTEGER,
  `createdAt` DATETIME NOT NULL,
  `updatedAt` DATETIME NOT NULL,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`listingId`)
  REFERENCES `listings` (`pledgeId`) ON DELETE NO ACTION ON UPDATE CASCADE) ENGINE=InnoDB;
